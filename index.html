function haversine(la1,lo1,la2,lo2){
  const R=6371, dLa=(la2-la1)*Math.PI/180, dLo=(lo2-lo1)*Math.PI/180;
  const a=Math.sin(dLa/2)**2+Math.cos(la1*Math.PI/180)*Math.cos(la2*Math.PI/180)*Math.sin(dLo/2)**2;
  return R*2*Math.atan2(Math.sqrt(a),Math.sqrt(1-a));
}

window.addItem=function(id){
  const p=PRODUCTS.find(x=>x.id===id);
  const ex=cart.find(x=>x.id===id);
  if(ex)ex.qty++;else cart.push({...p,qty:1});
  syncCart(); toast(`${p.name} added ✓`);
};
window.chgQty=function(id,d){
  const i=cart.find(x=>x.id===id);
  if(!i)return; i.qty+=d;
  if(i.qty<=0)cart=cart.filter(x=>x.id!==id);
  syncCart();
};
function syncCart(){

  const sub=cart.reduce((s,i)=>s+i.price*i.qty,0);
  const badge=document.getElementById('cartBadge');

  badge.classList.toggle('opacity-0',qty===0);
  badge.classList.toggle('scale-75',qty===0);
  document.getElementById('cartFee').textContent=deliveryFee===0?'FREE':'₹'+deliveryFee;
  document.getElementById('cartSub').textContent='₹'+sub;
  document.getElementById('cartTotal').textContent='₹'+(sub+deliveryFee);

  const foot=document.getElementById('cartFoot');

      <i class="fas fa-shopping-basket text-5xl mb-4"></i>
      <p class="font-medium text-gray-400">Your cart is empty</p>
      <p class="text-sm text-gray-300 mt-1">Browse products above</p></div>`;

      <div class="flex items-center gap-3 bg-white p-3 rounded-xl border border-gray-100 shadow-sm">
        <div class="w-14 h-14 rounded-xl overflow-hidden flex-shrink-0 img-bg">

    disableDefaultUI:false,
    zoomControl:true,
    mapTypeControl:false,
    streetViewControl:false,
    fullscreenControl:true,
    styles:mapStyle()
  });

    map:trackingMap,
    title:HUB_NAME,

  });
  new google.maps.InfoWindow({content:`<div style="font-family:Syne,sans-serif;font-weight:700;color:#ea580c;padding:4px 2px">📦 ${HUB_NAME}</div>`}).open(trackingMap,hubMark);

    title:'Your Delivery Location',
    icon:{url:'https://maps.google.com/mapfiles/ms/icons/green-dot.png',scaledSize:new google.maps.Size(36,36)}

  new google.maps.InfoWindow({content:`<div style="font-family:Syne,sans-serif;font-weight:700;color:#16a34a;padding:4px 2px">🏠 ${custAddr.split(',')[0]}</div>`}).open(trackingMap,destMark);

  const droneSVG = `<svg xmlns="http://www.w3.org/2000/svg" width="52" height="52" viewBox="0 0 52 52">
    <circle cx="26" cy="26" r="24" fill="rgba(249,115,22,0.15)" stroke="#f97316" stroke-width="1.5"/>

    <circle cx="44" cy="44" r="5" fill="#4b5563"/>
    <circle cx="26" cy="26" r="3" fill="white"/>

    position:{lat:HUB_LAT,lng:HUB_LNG},

    });

    // Draw leg 2: Store to Customer
    DS.route({origin:{lat:STORE_LAT,lng:STORE_LNG},destination:{lat:destLat,lng:destLng},travelMode:google.maps.TravelMode.DRIVING},(r2,s2)=>{
      const path2 = s2==='OK' ? r2.routes[0].overview_path : [{lat:STORE_LAT,lng:STORE_LNG},{lat:destLat,lng:destLng}].map(p=>new google.maps.LatLng(p.lat,p.lng));

      });

      // Combine both paths and animate drone through them
      const fullPath=[...path1,...path2];
      animateDrone(fullPath, path1.length);
    });
  });
}

function animateDrone(path, splitIdx){
  if(!droneMarker||!path.length)return;
  if(droneTimer)clearInterval(droneTimer);

  let idx=0;
  const total=path.length;

  function getLatLng(pt){
    return typeof pt.lat==='function' ? {lat:pt.lat(),lng:pt.lng()} : {lat:pt.lat,lng:pt.lng};
  }

  droneTimer=setInterval(()=>{
    if(idx>=total-1){
      clearInterval(droneTimer);
      // Arrived!
      document.getElementById('etaDisplay').textContent='✓';
      document.getElementById('trackStatus').textContent='Package delivered! 🎉';
      document.getElementById('step3').className='bg-white rounded-2xl p-4 shadow flex items-center gap-3 border-2 step-done';
      document.getElementById('step4').className='bg-white rounded-2xl p-4 shadow flex items-center gap-3 border-2 step-done';
      return;
    }
    idx++;
    const pos=getLatLng(path[idx]);
    droneMarker.setPosition(pos);

    // Smooth pan to keep drone centred
    trackingMap.panTo(pos);

    // Update ETA
    const remaining=total-idx;
    const secs=Math.round(remaining*0.25);
    const m=Math.floor(secs/60), s=secs%60;
    document.getElementById('etaDisplay').textContent=`${m}:${String(s).padStart(2,'0')}`;

    // Update status at midpoint
    if(idx===splitIdx){
      document.getElementById('trackStatus').textContent='Picked up from store — heading to you!';
    }
  },200); // 200ms per step = smooth glide
}

function buildLocMap(){
  if(typeof google==='undefined'){
    document.getElementById('locationMap').innerHTML=`<div class="w-full h-full flex items-center justify-center text-gray-400 flex-col gap-2"><i class="fas fa-exclamation-triangle text-3xl text-orange-300"></i><span class="text-sm">Maps API unavailable. Check API key.</span></div>`;
    return;
  }
  const iLat=custLat||23.2156, iLng=custLng||72.6369;
  locMap=new google.maps.Map(document.getElementById('locationMap'),{
    center:{lat:iLat,lng:iLng}, zoom:13,
    mapTypeControl:false, streetViewControl:false, fullscreenControl:false,
    styles:mapStyle()
  });

  // Hub pin (fixed)
  new google.maps.Marker({
    position:{lat:HUB_LAT,lng:HUB_LNG}, map:locMap, title:HUB_NAME,
    icon:{url:'https://maps.google.com/mapfiles/ms/icons/orange-dot.png',scaledSize:new google.maps.Size(32,32)}
  });

  // Draggable customer pin
  locMarker=new google.maps.Marker({
    position:{lat:iLat,lng:iLng}, map:locMap, draggable:true, title:'Your location',
    icon:{url:'https://maps.google.com/mapfiles/ms/icons/blue-dot.png',scaledSize:new google.maps.Size(36,36)}
  });

  const onPinMove=ll=>{
    const lat=ll.lat(), lng=ll.lng();
    pendingLoc={lat,lng};
    const km=haversine(HUB_LAT,HUB_LNG,lat,lng);
    showDistInfo(km);
    reverseGeo(lat,lng);
    document.getElementById('confirmLocBtn').disabled=false;
  };
  locMarker.addListener('dragend',e=>onPinMove(e.latLng));
  locMap.addListener('click',e=>{locMarker.setPosition(e.latLng);onPinMove(e.latLng)});

  // Autocomplete
  const inp=document.getElementById('locSearch');
  const ac=new google.maps.places.Autocomplete(inp,{
    componentRestrictions:{country:'in'},
    bounds:new google.maps.LatLngBounds({lat:22.9,lng:72.4},{lat:23.4,lng:72.9})
  });
  ac.addListener('place_changed',()=>{
    const pl=ac.getPlace();
    if(!pl.geometry)return;
    const lat=pl.geometry.location.lat(), lng=pl.geometry.location.lng();
    locMap.setCenter({lat,lng}); locMap.setZoom(15);
    locMarker.setPosition({lat,lng});
    pendingLoc={lat,lng};
    custAddr=pl.formatted_address||pl.name||'Selected location';
    showDistInfo(haversine(HUB_LAT,HUB_LNG,lat,lng));
    document.getElementById('confirmLocBtn').disabled=false;
  });

  if(custLat){pendingLoc={lat:custLat,lng:custLng};showDistInfo(distKm);document.getElementById('confirmLocBtn').disabled=false}
}

function reverseGeo(lat,lng){
  if(!google.maps.Geocoder)return;
  new google.maps.Geocoder().geocode({location:{lat,lng}},(r,s)=>{
    if(s==='OK'&&r[0])custAddr=r[0].formatted_address;
  });
}
